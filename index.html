Closures

(function(){
    var subscriptions = {};
    return {
        subscribe = function(event, handler){
            (subscriptions[event] = subscriptions[event] || [])
            .push(handler);
        },
        publish = function(event, data){
            (subscriptions[event] || [])
            .forEach(function(subscriber){
                subscriber(data);
            });
        }
    };
}());
